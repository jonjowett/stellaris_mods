endgame_unity_choose_influence = {
	length = 0
	is_ambition = yes
	ai_weight = { weight = 1 }
	potential = {
		has_technology = tech_ascension_theory # Must have the tech that unlocks unity ambitions
		NOT = { has_modifier = endgame_unity_influence_5 }

		if = {

		else_if = {
			limit = { has_modifier = endgame_unity_influence_2 }
			remove_modifier = endgame_unity_influence_2

		}
		else_if = {
			limit = { has_modifier = endgame_unity_influence_3 }
			remove_modifier = endgame_unity_influence_3
			add_modifier = { modifier = endgame_unity_influence_4 days = -1 }
		}
		else_if = {
			limit = { has_modifier = endgame_unity_influence_4 }
			remove_modifier = endgame_unity_influence_4
			add_modifier = { modifier = endgame_unity_influence_5 days = -1 }

			limit = { has_modifier = endgame_unity_roboticists_m_2 }
			remove_modifier = endgame_unity_roboticists_m_2
			add_modifier = { modifier = endgame_unity_roboticists_m_3 days = -1 }
			if = {
				limit = { NOT = { has_technology = tech_nanite_assemblers } }
				add_research_option = tech_nanite_assemblers
			}
		}
		else_if = {
			limit = { has_modifier = endgame_unity_roboticists_m_3 }
			remove_modifier = endgame_unity_roboticists_m_3
			add_modifier = { modifier = endgame_unity_roboticists_m_4 days = -1 }
		}
		else_if = {
			limit = { has_modifier = endgame_unity_roboticists_m_4 }
			remove_modifier = endgame_unity_roboticists_m_4
			add_modifier = { modifier = endgame_unity_roboticists_m_5 days = -1 }
		}
		else = {
			add_modifier = { modifier = endgame_unity_roboticists_m_1 days = -1 }
			if = {
				limit = { NOT = { has_technology = tech_binary_motivators } }
				add_research_option = tech_binary_motivators
			}
		}
		
		#Decrease in species modification cost is granted separately because all types of species modification are controlled by the same variable.
		if = {
			limit = { has_modifier = endgame_unity_decrease_species_modification_cost_1 }
			remove_modifier = endgame_unity_decrease_species_modification_cost_1
			add_modifier = { modifier = endgame_unity_decrease_species_modification_cost_2 days = -1 }
		}
		else_if = {
			limit = { has_modifier = endgame_unity_decrease_species_modification_cost_2 }
			remove_modifier = endgame_unity_decrease_species_modification_cost_2
			add_modifier = { modifier = endgame_unity_decrease_species_modification_cost_3 days = -1 }
		}
		else_if = {
			limit = { has_modifier = endgame_unity_decrease_species_modification_cost_3 }
			remove_modifier = endgame_unity_decrease_species_modification_cost_3
			add_modifier = { modifier = endgame_unity_decrease_species_modification_cost_4 days = -1 }
		}
		else_if = {
			limit = { has_modifier = endgame_unity_decrease_species_modification_cost_4 }
			remove_modifier = endgame_unity_decrease_species_modification_cost_4
			add_modifier = { modifier = endgame_unity_decrease_species_modification_cost_5 days = -1 }
		}
		else_if = {
			limit = { has_modifier = endgame_unity_decrease_species_modification_cost_5 }
			#Capped-out - do nothing
		}
		else = {
			add_modifier = { modifier = endgame_unity_decrease_species_modification_cost_1 days = -1 }
		}
		
	}
